class A is
    method F(x: Integer) : Integer is
        return x
    end
end

class B extends A is
    // OK
    method F(x: Integer) : Integer is
        return x.UnaryMinus()
    end

    // Ошибка: другой тип параметра
    method F(x: Real) : Integer is
        return Integer(0)
    end

    // Ошибка: другой тип возврата
    method G(x: Integer) : Real is
        return x.toReal()
    end
end
