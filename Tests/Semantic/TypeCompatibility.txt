class TypeCompatibilityTest is
    this is

        var intVar : 10
        var realVar : 3.14
        var boolVar : true


        realVar := intVar.txReal() // OK: Integer to Real via method call (widening)
        intVar := realVar // ERROR: Cannot assign Real to Integer.

        var text : "hello" // ERROR: No string type.
        var flag : "true" // ERROR: No string type, cannot assign to Boolean.

        // VALID & INVALID: Method return assignments
        var floatResult : getReal() // OK: Method returns Real, variable is Real.
        // var intResult : getReal() // ERROR: Cannot assign Real to Integer.

        // Expressions are method calls
        var a : 5
        var b : 2.0
        "float result1 = a + b" becomes:
        var result1 : a.Plus(b) // OK: a.Plus(b) returns a Real

        "int result2 = a + b" becomes:
        var result2 : a.Plus(b) // ERROR: a.Plus(b) returns Real, cannot assign to Integer variable.

        // Invalid operations
        var flag : true
        "flag + 5" becomes:
        var invalid : flag.Plus(5) // ERROR: Class 'Boolean' does not have a method Plus(Integer).

    end

    // Method declaration.
    method getReal : Real is
        return 2.5
    end

end
