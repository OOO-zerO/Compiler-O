class ControlFlowTest is
    method complexIfElse(x: Integer, y: Integer) is
        if x.Greater(y) then
            var result : 1
        else
            // Using a nested 'if' for 'else if'
            if x.Less(y) then
                var result : 2
            else
                var result : 3
            end
        end
    end

    method nestedLoops is
        var i : 0
        while i.Less(5) loop
            var j : 0
            while j.Less(3) loop
                var dummy : i.Plus(j)
                j := j.Plus(1)
            end
            i := i.Plus(1)
        end
    end

    method switchLike(value: Integer) is
        if value.Equal(1) then
            var result : 1
        else
            if value.Equal(2) then
                var result : 2
            else
                if value.Equal(3) then
                    var result : 3
                else
                    var result : 0 // Default case
                end
            end
        end
    end

end
