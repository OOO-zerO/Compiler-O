// Complex equations and nested blocks

class ComplexExpressions is
    method calculate(a: Integer, b: Integer, c: Integer) : Integer is
        // Prioritize operations using method calls
        var temp1 : Integer(a.Plus(b))
        var temp2 : Integer(temp1.Mult(c))
        var temp3 : Integer(a.Div(b))
        var result : Integer(temp2.Minus(temp3))
        
        if result.Greater(Integer(100)).And(result.Less(Integer(200))) then
            while result.Greater(Integer(0)) loop
                result := result.Minus(Integer(1))
                
                if result.Rem(Integer(2)).Equal(Integer(0)) then
                    // write("Even: " + result) - string ops not available
                end
            end
        end
        
        return result
    end
end
